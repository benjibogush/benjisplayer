# check version of react sripts
> npm info react-scripts version

# in package.json, change version to
>"react-scripts": "^3.4.1" (or the latest available version)

# install dependencies needed
> npm i @material-ui/core @material-ui/icons react react-dom react-scripts latest

# install dev dependencies 
> npm i --save-dev @types/react @types/react-dom typescript latest
